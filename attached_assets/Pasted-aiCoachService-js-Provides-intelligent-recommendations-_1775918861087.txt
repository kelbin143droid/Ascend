📊 Habit Service
📄 habitService.js
Handles storage and persistence.
JavaScript
const HABITS_KEY = "habits";
const BAD_HABITS_KEY = "badHabits";

export const habitService = {
  getHabits() {
    return JSON.parse(localStorage.getItem(HABITS_KEY)) || [];
  },

  saveHabit(habit) {
    const habits = this.getHabits();
    habits.push({ ...habit, id: Date.now(), streak: 0 });
    localStorage.setItem(HABITS_KEY, JSON.stringify(habits));
  },

  getBadHabits() {
    return JSON.parse(localStorage.getItem(BAD_HABITS_KEY)) || [];
  },

  saveBadHabit(habit) {
    const habits = this.getBadHabits();
    habits.push({ ...habit, id: Date.now(), resistanceStreak: 0 });
    localStorage.setItem(BAD_HABITS_KEY, JSON.stringify(habits));
  }
};
➕ Add Habit Modal
📄 AddHabitModal.jsx
JavaScript
import React, { useState } from "react";
import { habitService } from "../../services/habitService";
import { aiCoachService } from "../../services/aiCoachService";
import "../../styles/habits.css";

const AddHabitModal = ({ onClose, onSave }) => {
  const [habit, setHabit] = useState({
    name: "",
    category: "focus",
    duration: 5,
    frequency: "daily",
    xp: 5,
    hp: 0,
    mp: 0,
    time: ""
  });

  const [aiTip, setAiTip] = useState("");

  const handleCategoryChange = (category) => {
    const recommendation =
      aiCoachService.getHabitRecommendation(category);

    if (recommendation) {
      setHabit({
        ...habit,
        name: recommendation.name,
        duration: recommendation.duration,
        xp: recommendation.xp,
        hp: recommendation.hp || 0,
        mp: recommendation.mp || 0,
        category
      });
      setAiTip(recommendation.message);
    }
  };

  const handleSubmit = () => {
    habitService.saveHabit(habit);
    onSave();
    onClose();
  };

  return (
    <div className="modal-overlay">
      <div className="modal-card">
        <h2>Add New Habit</h2>

        <label>Category</label>
        <select
          onChange={(e) => handleCategoryChange(e.target.value)}
        >
          <option value="focus">Focus</option>
          <option value="vitality">Vitality</option>
          <option value="strength">Strength</option>
          <option value="agility">Agility</option>
        </select>

        <label>Habit Name</label>
        <input
          type="text"
          value={habit.name}
          onChange={(e) =>
            setHabit({ ...habit, name: e.target.value })
          }
        />

        <label>Duration (minutes)</label>
        <input
          type="number"
          value={habit.duration}
          onChange={(e) =>
            setHabit({ ...habit, duration: e.target.value })
          }
        />

        <label>Time</label>
        <input
          type="time"
          onChange={(e) =>
            setHabit({ ...habit, time: e.target.value })
          }
        />

        {aiTip && (
          <div className="ai-tip">
            🤖 AI Coach: {aiTip}
          </div>
        )}

        <div className="modal-actions">
          <button onClick={onClose}>Cancel</button>
          <button className="primary-btn" onClick={handleSubmit}>
            Save Habit
          </button>
        </div>
      </div>
    </div>
  );
};

export default AddHabitModal;
🚫 Break Bad Habit Modal
📄 BreakHabitModal.jsx
JavaScript
import React, { useState } from "react";
import { habitService } from "../../services/habitService";
import { aiCoachService } from "../../services/aiCoachService";
import "../../styles/habits.css";

const BreakHabitModal = ({ onClose, onSave }) => {
  const [habitName, setHabitName] = useState("");
  const [solution, setSolution] = useState(null);

  const handleAnalyze = () => {
    const result =
      aiCoachService.getBadHabitSolution(habitName);
    setSolution(result);
  };

  const handleSubmit = () => {
    habitService.saveBadHabit({
      name: habitName,
      replacement: solution.replacement,
      tip: solution.tip
    });

    onSave();
    onClose();
  };

  return (
    <div className="modal-overlay">
      <div className="modal-card">
        <h2>Break a Bad Habit</h2>

        <label>Bad Habit</label>
        <input
          type="text"
          placeholder="e.g., Late Night Scrolling"
          onChange={(e) => setHabitName(e.target.value)}
        />

        <button className="secondary-btn" onClick={handleAnalyze}>
          Analyze with AI Coach
        </button>

        {solution && (
          <div className="ai-tip">
            <p><strong>Replacement:</strong> {solution.replacement}</p>
            <p><strong>Tip:</strong> {solution.tip}</p>
          </div>
        )}

        <div className="modal-actions">
          <button onClick={onClose}>Cancel</button>
          <button
            className="primary-btn"
            onClick={handleSubmit}
            disabled={!solution}
          >
            Save Plan
          </button>
        </div>
      </div>
    </div>
  );
};

export default BreakHabitModal;

import React, { useState } from "react";
import AddHabitModal from "./AddHabitModal";
import BreakHabitModal from "./BreakHabitModal";

const Habits = () => {
  const [showAdd, setShowAdd] = useState(false);
  const [showBreak, setShowBreak] = useState(false);

  return (
    <div className="habits-container">
      <h1>HABITS</h1>

      <button onClick={() => setShowAdd(true)}>
        + Add Habit
      </button>

      <button onClick={() => setShowBreak(true)}>
        Break a Bad Habit
      </button>

      {showAdd && (
        <AddHabitModal
          onClose={() => setShowAdd(false)}
          onSave={() => window.location.reload()}
        />
      )}

      {showBreak && (
        <BreakHabitModal
          onClose={() => setShowBreak(false)}
          onSave={() => window.location.reload()}
        />
      )}
    </div>
  );
};

export default Habits;

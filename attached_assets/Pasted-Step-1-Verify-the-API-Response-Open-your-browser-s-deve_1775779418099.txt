Ensure GuidedSessionPage.tsx writes to session storage only once.

const hasRecordedCompletion = useRef(false);

const recordCompletion = (day: number) => {
  if (hasRecordedCompletion.current) return;
  hasRecordedCompletion.current = true;

  sessionStorage.setItem(
    "ascend_just_completed_day",
    String(day)
  );
};

🔧 Step 6: Correct Backend Onboarding Calculation
Replace your onboarding logic in server/routes.ts with:
TypeScript
const TOTAL_ONBOARDING_DAYS = 5;

const completedDays: number[] = [];

for (const [habitId, day] of Object.entries(ONBOARDING_HABIT_TO_DAY)) {
  if (allCompletions.some(c => c.habitId === habitId)) {
    completedDays.push(day);
  }
}

completedDays.sort((a, b) => a - b);

const onboardingDay = Math.min(
  completedDays.length + 1,
  TOTAL_ONBOARDING_DAYS
);

const isOnboardingComplete =
  completedDays.length >= TOTAL_ONBOARDING_DAYS;
Return:
TypeScript
res.json({
  onboardingDay,
  completedDays,
  isOnboardingComplete,
  streak,
  hasCompletedHabitToday,
});
